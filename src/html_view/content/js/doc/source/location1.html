Ext.ns("Philosophy.model");

/**

 * @extends Philosophy.model.Entity
 * 
 * Philosphy Location Proxy
 * 
 * @author rbarriga

 * @copyright (c) 2010, by IKOM
 * @date      15. July 2010
 */
Philosophy.model.Location = Ext.extend(Philosophy.model.Entity, {
    createURL: Phi.UriTemplate.getUri('locationService', 'locationCreate'),
    readURL: Phi.UriTemplate.getUri('locationService', 'location'),
    updateURL: Phi.UriTemplate.getUri('locationService', 'locationUpdate'),
    removeURL: Phi.UriTemplate.getUri('locationService', 'location'),

    initComponent: function (config) {
        this.addEvents(
            'enabled',
            'favorite'
		);
        Philosophy.model.Location.superclass.initComponent.call(config);
    }
    ,
    favorite: function (o) {
        Ext.Ajax.request({
            url: Phi.UriTemplate.getUri('locationService', 'locationFavorite'),
            method: 'PUT',
            headers: { 'Content-Type': 'text/json' },
            jsonData: o,
            scope:this,
            success: function (response, options) {
                this.fireEvent('favorite', response);
            }
        });     
    }
});
