/**

Phi.view.panel.Search = Ext.extend(Ext.Panel, {
    title: Philosophy.Globalization.For('Search'),
    layout: 'fit',
    border: false,
    collapsible: true,
    autoscroll: true,
    plugins: [Ext.ux.plugins.ToggleCollapsible],
    searchEngine: Philosophy.SearchEngines,

    initComponent: function() {
        var _this = this;
        var tbar = new Ext.Toolbar({
            items: [{ xtype: 'tbfill' },
                    {
                        iconCls: 'icon-clear',
                        text: Philosophy.Globalization.For('Clear'),
                        tooltip: Philosophy.Globalization.For('Clear'),
                        handler: function() { _this.clear(); }
                    }]
        });
        this.tbar = tbar;
        Phi.view.panel.Search.superclass.initComponent.call(this);
    }
    ,
    clear: function() {
        this.body.dom.innerHTML = "";
        this.body.dom.className = "main_container";
        this.doLayout();
        this.body.highlight('#f6e8b0', { block: true });
    }
});  // eof Phi.view.panel.Search
// eof
