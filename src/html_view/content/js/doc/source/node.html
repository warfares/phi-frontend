Philosophy.model.Node = Ext.extend(Ext.util.Observable, {
    constructor: function (config) {
        this.addEvents(
			'getnodes'
		);
        Philosophy.model.Node.superclass.constructor.call(config);
    }
	,
    getNodes: function () {
        Ext.Ajax.request({
            url: Phi.UriTemplate.getUri('utilService', 'nodeGetNodes'),
            method: 'GET',
            headers: { 'Content-Type': 'text/json' },
            scope: this,
            success: function (response, options) {
                var n = Ext.util.JSON.decode(response.responseText);
                var nodes = [];
                nodes.push(n);
                this.fireEvent('getnodes', nodes);
            }
        });    
    }
});
