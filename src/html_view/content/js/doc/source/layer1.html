Ext.ns("Philosophy.model");

/**
 * @class Philosophy.model.Layer 
 * @extends Philosophy.model.Entity
 * 
 * Philosphy Location Proxy
 * 
 * @author rbarriga

 * @copyright (c) 2010, by IKOM
 * @date      15. July 2010
 */
Philosophy.model.Layer = Ext.extend(Philosophy.model.Entity, {
    readURL: Philosophy.UriTemplate.getUri('layerService', 'layerRead'),

    initComponent: function (config) {
        Philosophy.model.Layer.superclass.initComponent.call(config);
    }
    ,
    
/**
    * Read entity
    * @param {String} entity identification (key)
    */
    read: function (id) {
        var p = {
            name: id
        };
        Ext.Ajax.request({
            url: this.readURL + '?' + Ext.urlEncode(p),
            scope: this,
            success: function (r, opt) {
                var o = Ext.util.JSON.decode(r.responseText);
                this.fireEvent('read', o);
            }
        });
    }
});
