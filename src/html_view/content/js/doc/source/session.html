Ext.ns("Philosophy");

/**
* @class Philosophy.Session
* 
* Philosophy current logged user 
* 
* @author rbarriga
* @version 1.2
* @copyright (c) 2010, by IKOM
* @date      21. August 2010
* @singleton
*/
Philosophy.Session = {
    user: null,
    setUser: function (user) {
        this.user = user;
    }
    ,
    getUser: function () {
        return this.user.UserName;
    }
    ,
    isRole: function (role) {
        var result = false;
        Ext.each(this.user.RolesVO, function (r) {
            if (r.Id === role)
                result = true;
        }, this);
        return result;
    }
}; // eo Philosophy.Session
// eof
